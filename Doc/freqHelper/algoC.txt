#define CST_MHZ_TO_HZ                       (1000000U)
#define CST_MAX_UINT_16BIT                 (65535U)
#define CST_MAX_ARR_TARGET                 (50000U)
#define FMKTIM_ARR_DECREASING_VAL_16B      (18000U)
#define FMKTIM_FREQ_COMPUTE_DELTA_ACCEPTANCE (1.0f)
#define FMKTIM_LOOP_MAX                     (5U)  // Limite de sécurité pour les boucles
typedef struct
{
    t_uint16 prescaler_u16;
    t_uint16 arr_u16;
    t_float32 realFreq_f32;
    t_float32 deltaFreq_f32;
} t_sPWMConfig;

t_float32 max_delta_freq_f32 = 0.0f;
t_uint32 g_frequency_u32;
/* Fonction pour calculer la fréquence PWM à partir du prescaler et ARR */
static t_float32 s_ComputePwmFreq(t_uint32 timerFreqMHz_u32, t_uint16 prescaler_u16, t_uint16 arr_u16)
{
    t_float32 timerClkHz = (t_float32)(timerFreqMHz_u32 * CST_MHZ_TO_HZ);
    return timerClkHz / ((prescaler_u16 + 1) * (arr_u16 + 1));
}

/* Fonction principale pour calculer le prescaler et l'ARR à partir de la fréquence PWM cible */
t_sPWMConfig FMKTIM_ComputePscFromFreq(t_uint32 timerFreqMHz_u32, t_float32 pwmFreqTarget_f32)
{
    t_sPWMConfig config_s = {0};
    t_float32 ratio_f32 = (t_float32)(timerFreqMHz_u32 * CST_MHZ_TO_HZ) / pwmFreqTarget_f32;

    t_uint32 targetArr_u32 = CST_MAX_ARR_TARGET;
    t_float32 pscTheo_f32 = 0.0f;
    t_uint16 realPsc_u16 = 0;
    t_float32 deltaPsc_f32 = 0.0f;
    t_uint32 realArr = 0;
    t_uint32 deltaArr = 0;
    t_uint32 loopCnt = 0;

    // === Étape 1 : Calcul prescaler théorique avec ajustement de ARR
    do
    {
        pscTheo_f32 = (ratio_f32 / (t_float32)targetArr_u32) - 1.0f;
        loopCnt++;
        if (loopCnt >= FMKTIM_LOOP_MAX)
        {
            printf("[ERROR] Boucle ARR trop longue.\n");
            break;
        }
        if (pscTheo_f32 > (t_float32)CST_MAX_UINT_16BIT)
        {
            targetArr_u32 -= FMKTIM_ARR_DECREASING_VAL_16B;
        }
    } while (pscTheo_f32 > (t_float32)CST_MAX_UINT_16BIT);

    realPsc_u16 = (t_uint16)pscTheo_f32;
    deltaPsc_f32 = pscTheo_f32 - (t_float32)realPsc_u16;

    if (realPsc_u16 > 0)
    {
        loopCnt = 0;
        do
        {
            deltaArr = (t_uint32)roundf(((targetArr_u32 * (deltaPsc_f32 + 1.0f)) - realPsc_u16) / realPsc_u16);
            realArr = (t_uint32)(targetArr_u32 + deltaArr);
            loopCnt++;
            if (loopCnt >= FMKTIM_LOOP_MAX)
            {
                printf("[ERROR] Boucle realArr trop longue.\n");
                break;
            }
            if (realArr >= CST_MAX_UINT_16BIT)
            {
                targetArr_u32 -= FMKTIM_ARR_DECREASING_VAL_16B;
                pscTheo_f32 = (ratio_f32 / targetArr_u32) - 1.0f;
                realPsc_u16 = (t_uint16)pscTheo_f32;
               deltaPsc_f32 = pscTheo_f32 - (t_float32)realPsc_u16;
            }
        } while (realArr >= CST_MAX_UINT_16BIT);

        config_s.prescaler_u16 = realPsc_u16 - 1U;
        config_s.arr_u16 = (t_uint16)realArr;
    }
    else
    {
        realPsc_u16 = 0;
        loopCnt = 0;
        do
        {
            realArr = (t_uint32)(((t_float32)(timerFreqMHz_u32 * CST_MHZ_TO_HZ) /
                        (pwmFreqTarget_f32 * (realPsc_u16 + 1U))) - 1.0f);
            if (realArr >= CST_MAX_UINT_16BIT)
            {
                realPsc_u16++;
            }
            loopCnt++;
            if (loopCnt >= FMKTIM_LOOP_MAX)
            {
                printf("[ERROR] Boucle realArr (cas PSC=0) trop longue.\n");
                break;
            }
        } while (realArr >= CST_MAX_UINT_16BIT);

        config_s.prescaler_u16 = realPsc_u16;
        config_s.arr_u16 = (t_uint16)realArr;
    }

    config_s.realFreq_f32 = s_ComputePwmFreq(timerFreqMHz_u32,
                                             config_s.prescaler_u16,
                                             config_s.arr_u16);
    config_s.deltaFreq_f32 = fabsf(pwmFreqTarget_f32 - config_s.realFreq_f32);
    t_float32 ration_freq = config_s.deltaFreq_f32 / pwmFreqTarget_f32;
    
    if(ration_freq > max_delta_freq_f32)
    {
        max_delta_freq_f32 = ration_freq;
    }
    if (   ration_freq > 0.05f
        || config_s.arr_u16 >= CST_MAX_UINT_16BIT
        || config_s.prescaler_u16 >= CST_MAX_UINT_16BIT)
    {
  
            max_delta_freq_f32 = config_s.deltaFreq_f32;
            g_frequency_u32 = (t_uint32)pwmFreqTarget_f32;

        
        /*printf("[WARN] Target %.2f Hz -> Real %.2f Hz | ARR %u | PSC %u | Δ%.4f\n",
               pwmFreqTarget_f32,
               config_s.realFreq_f32,
               config_s.arr_u16,
               config_s.prescaler_u16,
               config_s.deltaFreq_f32);*/
    }

    return config_s;
}
